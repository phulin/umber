% Fixed page-builder, insertion, mark, and output-routine workload.
\input benchmark-preamble.inc
\hsize=24pc
\vsize=36pc
\maxdepth=4pt
\countdef\pagesmade=100
\countdef\blocks=101
\chardef\benchmarkinsert=254
\pagesmade=0 \blocks=0
\count254=1000
\dimen254=18pc
\skip254=0pt plus 1fil

\output={%
  \global\advance\pagesmade by 1
  \shipout\vbox{%
    \hbox to\hsize{\benchmarkrm Page \the\pagesmade\hfil
      \firstmark--\botmark}%
    \benchmarkoffinterlineskip\hrule
    \vskip6pt
    \box255
  }%
}

\def\block{%
  \mark{block-\the\blocks}%
  \hbox to\hsize{\benchmarkrm page-builder material \the\blocks\hfil
    \vrule height7pt depth2pt width12pt}%
  \benchmarknobreak\vskip3pt plus5pt minus2pt
  \penalty50
  \hbox to\hsize{\benchmarkit competing vertical breakpoints\hfil
    \benchmarkbf fixed work}%
  \vskip5pt plus3pt minus1pt
}

\def\work{%
  \advance\blocks by 1
  \block
  \ifnum\blocks=10
    \insert\benchmarkinsert{\hsize=18pc\benchmarkrm insertion material
      \number\pagesmade\par}%
    \blocks=0
  \fi
  \ifnum\pagesmade<500 \expandafter\work\fi
}
\work
\benchmarkeject
\message{BENCH pages \the\pagesmade}
\end
