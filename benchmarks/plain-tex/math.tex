% Fixed math-list construction and packing workload.
\input benchmark-preamble.inc
\countdef\iterations=100
\dimendef\checksum=100
\chardef\formula=100
\iterations=0 \checksum=0pt

\def\makeformula{%
  \setbox\formula=\hbox{$\displaystyle
    \left(\mathop\BenchmarkSum\limits_{i=1}^{n}
      {x_i^2+\BenchmarkRadical{1+x_i}\over 1+x_i^2}\right)^{1/2}
    +{\mathop\BenchmarkProd\limits_{j=1}^{m}(1+y_j)\over
       \BenchmarkRadical{\left|\BenchmarkAlpha-\BenchmarkBeta\right|
         +{1\over 1+\BenchmarkGamma^2}}}
    +\mathop{l i m}\limits_{k\mathrel\BenchmarkArrow\BenchmarkInfinity}
       \left(1+{a_k\over b_k+{c_k\over d_k}}\right)
  $}%
  \global\checksum=\wd\formula
}

\def\work{%
  \advance\iterations by 1
  \makeformula
  \ifnum\iterations<20000 \expandafter\work\fi
}
\work
\ifdim\checksum=0pt \errmessage{math checksum mismatch}\fi
\message{BENCH math \the\iterations}
\copy\formula
\benchmarkbye
